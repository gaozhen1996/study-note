# 启动mysql

方式一：

sudo /etc/init.d/mysql start  

方式二：

sudo start mysql 

方式三：

sudo service mysql start

# 停止mysql

方式一：

sudo /etc/init.d/mysql stop  

方式二：

sudo stop mysql 

方式三：

sudo service mysql stop

# 重启mysql

方式一：

sudo/etc/init.d/mysql restart

方式二：

sudo restart mysql

方式三：

sudo service mysql restart
